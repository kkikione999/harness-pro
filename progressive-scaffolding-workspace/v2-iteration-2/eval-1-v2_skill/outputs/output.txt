```typescript
// [HARNESS-INJECT-START: structured_log]
// Tier 2: Structured JSON logging — enables O1 Semantic Logs + O2 Correlation Propagation
// Activates when HARNESS_CONTROL=1
export interface HarnessLogEntry {
  level: 'INFO' | 'WARN' | 'ERROR' | 'DEBUG'
  timestamp: string
  correlationId: string
  component: string
  message: string
  metadata: Record<string, unknown>
}

export function harnessLog(
  level: 'INFO' | 'WARN' | 'ERROR' | 'DEBUG',
  message: string,
  metadata: Record<string, unknown> = {},
  component = 'app'
): HarnessLogEntry {
  const entry: HarnessLogEntry = {
    level,
    message,
    metadata,
    component,
    timestamp: new Date().toISOString(),
    correlationId: process.env.HARNESS_CORRELATION_ID || '',
  }
  process.stdout.write(JSON.stringify(entry) + '\n')
  return entry
}

export function harnessInfo(msg: string, meta: Record<string, unknown> = {}): HarnessLogEntry {
  return harnessLog('INFO', msg, meta)
}
export function harnessError(msg: string, meta: Record<string, unknown> = {}): HarnessLogEntry {
  return harnessLog('ERROR', msg, meta)
}
export function harnessWarn(msg: string, meta: Record<string, unknown> = {}): HarnessLogEntry {
  return harnessLog('WARN', msg, meta)
}
export function harnessDebug(msg: string, meta: Record<string, unknown> = {}): HarnessLogEntry {
  return harnessLog('DEBUG', msg, meta)
}
// [HARNESS-INJECT-END: structured_log]
```
